@lru_cache()  # This is a decorator
def fib(n):
    if n <= 1:
        return 1
    return fib(n - 1) + fib(n - 2)

The solution

>>> def all_files():
...     return set(os.listdir('.'))
...
>>> before = all_files()
>>> while True:
...     time.sleep(0.5)
...     new = all_files()
...     change = new - before
...     for f in change:
...         print(f)
...
...     before = new
